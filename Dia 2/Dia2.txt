----------------Conversiones de tipos de datos--------------
/*existen dos tipos de conversiones las implicitas y las explicitas*/
/*la implicita es realizada automaticamente por python sin que nos demos cuantes como el caso del input*/
/*la explicita ocurre cuando nosotros explicitamente hacemos una conversion mediante codigo para que un tipo de datos se pueda converitr a otro tipo de datos*/
/*usando este codigo podmeos que ver que num1 es un tipo int y num2 es de tipo float*/
num1 = 20
num2 = 29.6
print(type(num1))
print(type(num2))
/*ahora vamos a tranformar num1 con una operacion*/
num1 = 20
num2 = 29.6
num1 = num1 + num2

print(type(num1))
print(type(num2))
/*al ver los tipos de datos vemos que son ambos float, este es un tipo de conversion implicita ya que al hacer la operacion pytho lo que hizo fue convertir el numero de int a float automaticamente sin que nosotros pidieras explicitamente que lo convirtiera*/


/*ahora vamos a vonvertirlo a int el float que tenemos y eso se hace con la funicona de int()*/

print(f"{x} por {y} es igual a {x*y}")#multiplicacion
print(f"{x} dividido {y} es igual a {x/y}")#division

print(f"{z} dividido {y} es igual a {z//y}")#division al piso#al utilizar el // le decimos a python que no queremos decimales con la respuesta solo el resultado de la division al piso

print(f"{z} dividido {y} es igual a {z**y}")#potenciacion

-----redondear-----
/*la funciona round se utiliza para redondear el numero en dado caso que el resultado tenga decimales*/
resultado = round(90/7)
print(resultado)

/*la funcion round puede recibir dos parametros el primero parametro que el el valor que queremos redondear y el segundo parametro seria la cantidad de decimales que quiero para ese redondeo*/
resultado = round(90/7,3)
print(resultado)
